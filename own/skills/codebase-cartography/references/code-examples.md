# Codebase Cartography — Code Examples & Example Outputs

## generate-codemap.sh — Full Bash Implementation

```bash
#!/usr/bin/env bash
# generate-codemap.sh — Auto-generate codemap.md for AI context
# Usage: ./generate-codemap.sh [project-dir] [output-file]
# Requires: fd (or find), jq (optional), rg (optional)
set -euo pipefail

PROJECT_DIR="${1:-.}"
OUTPUT="${2:-${PROJECT_DIR}/codemap.md}"
DATE=$(date +%Y-%m-%d)

has_cmd() { command -v "$1" &>/dev/null; }

tree_cmd() {
  local dir="$1" depth="${2:-3}"
  if has_cmd fd; then
    fd --base-directory "$dir" --max-depth "$depth" \
       -E node_modules -E .git -E __pycache__ -E .next -E target -E build -E dist -E .venv \
       --strip-cwd-prefix 2>/dev/null | sort
  else
    find "$dir" -maxdepth "$depth" \
      -not -path '*/node_modules/*' -not -path '*/.git/*' -not -path '*/__pycache__/*' \
      -not -path '*/.next/*' -not -path '*/target/*' -not -path '*/build/*' \
      -not -path '*/dist/*' -not -path '*/.venv/*' -printf '%P\n' 2>/dev/null | sort
  fi
}

count_files() {
  local dir="$1"
  if has_cmd fd; then
    fd --type f --base-directory "$dir" -E node_modules -E .git -E __pycache__ -E .next -E target -E build -E dist -E .venv 2>/dev/null | wc -l
  else
    find "$dir" -type f -not -path '*/node_modules/*' -not -path '*/.git/*' -not -path '*/__pycache__/*' -not -path '*/.next/*' -not -path '*/target/*' -not -path '*/build/*' -not -path '*/dist/*' -not -path '*/.venv/*' 2>/dev/null | wc -l
  fi
}

detect_language() {
  local dir="$1" langs=()
  [ -f "$dir/tsconfig.json" ] && langs+=("TypeScript")
  [ -f "$dir/go.mod" ] && langs+=("Go")
  [ -f "$dir/Cargo.toml" ] && langs+=("Rust")
  [ -f "$dir/pom.xml" ] || [ -f "$dir/build.gradle" ] && langs+=("Java")
  [ -f "$dir/pyproject.toml" ] || [ -f "$dir/requirements.txt" ] && langs+=("Python")
  [ ${#langs[@]} -eq 0 ] && echo "Unknown" || printf '%s\n' "${langs[@]}" | paste -sd ', ' -
}

detect_framework() {
  local dir="$1"
  [ -f "$dir/next.config.ts" ] || [ -f "$dir/next.config.js" ] && echo "Next.js" && return
  [ -f "$dir/angular.json" ] && echo "Angular" && return
  [ -f "$dir/manage.py" ] && echo "Django" && return
  [ -f "$dir/go.mod" ] && echo "Go" && return
  [ -f "$dir/Cargo.toml" ] && echo "Rust" && return
  echo "None detected"
}

detect_package_manager() {
  local dir="$1"
  [ -f "$dir/pnpm-lock.yaml" ] && echo "pnpm" && return
  [ -f "$dir/yarn.lock" ] && echo "yarn" && return
  [ -f "$dir/bun.lockb" ] && echo "bun" && return
  [ -f "$dir/package-lock.json" ] && echo "npm" && return
  [ -f "$dir/go.sum" ] && echo "go modules" && return
  [ -f "$dir/Cargo.lock" ] && echo "cargo" && return
  echo "unknown"
}

detect_architecture_pattern() {
  local dir="$1"
  [ -d "$dir/src/domain" ] && [ -d "$dir/src/application" ] && echo "Hexagonal / Clean Architecture" && return
  [ -d "$dir/src/features" ] || [ -d "$dir/app/features" ] && echo "Feature-based modular" && return
  [ -d "$dir/cmd" ] && [ -d "$dir/internal" ] && echo "Standard Go Project Layout" && return
  echo "Flat / Convention-based"
}

extract_dependencies() {
  local dir="$1"
  if [ -f "$dir/package.json" ] && has_cmd jq; then
    jq -r '.dependencies // {} | to_entries[:10][] | "- **\(.key)** \(.value)"' "$dir/package.json" 2>/dev/null && return
  fi
  [ -f "$dir/go.mod" ] && grep -E '^\t[a-zA-Z]' "$dir/go.mod" 2>/dev/null | head -10 | awk '{print "- **" $1 "** " $2}' && return
  echo "- (Could not auto-detect dependencies)"
}

find_entry_points() {
  local dir="$1"
  [ -f "$dir/app/layout.tsx" ] && echo "- \`app/layout.tsx\` — Root layout"
  [ -f "$dir/app/page.tsx" ] && echo "- \`app/page.tsx\` — Home route"
  [ -f "$dir/src/index.ts" ] && echo "- \`src/index.ts\` — Main entry"
  [ -f "$dir/src/main.ts" ] && echo "- \`src/main.ts\` — Main entry"
  [ -f "$dir/main.go" ] && echo "- \`main.go\` — Main entry"
  [ -f "$dir/manage.py" ] && echo "- \`manage.py\` — Django management"
  [ -f "$dir/src/main.rs" ] && echo "- \`src/main.rs\` — Binary entry"
  for cfg in Dockerfile docker-compose.yml Makefile; do
    [ -f "$dir/$cfg" ] && echo "- \`$cfg\` — Build/deploy config"
  done
}

dir=$(cd "$PROJECT_DIR" && pwd)
name=$(basename "$dir")
file_count=$(count_files "$dir")
depth=3; [ "$file_count" -gt 1000 ] && depth=2; [ "$file_count" -lt 100 ] && depth=5

cat > "$OUTPUT" <<CODEMAP
# Codebase Map

> Auto-generated by codebase-cartography. Last updated: ${DATE}

## Project Overview

- **Name**: ${name}
- **Language**: $(detect_language "$dir")
- **Framework**: $(detect_framework "$dir")
- **Package Manager**: $(detect_package_manager "$dir")
- **Total Files**: ${file_count}

## Architecture

**Pattern**: $(detect_architecture_pattern "$dir")

## Directory Structure

\`\`\`
$(tree_cmd "$dir" "$depth" | head -80)
\`\`\`

## Entry Points

$(find_entry_points "$dir")

## Dependencies (Top 10)

$(extract_dependencies "$dir")
CODEMAP

echo "Done! codemap.md written to $OUTPUT"
echo "Token estimate: ~$(wc -w < "$OUTPUT") words"
```

## Quick Codemap (fd + rg)

```bash
#!/usr/bin/env bash
# quick-codemap.sh — Minimal codemap generator
set -euo pipefail
DIR="${1:-.}"
NAME=$(basename "$(cd "$DIR" && pwd)")
echo "# Codebase Map — ${NAME}"
echo "> Generated: $(date +%Y-%m-%d)"
echo ""
echo "## Structure"
echo '```'
fd --base-directory "$DIR" --max-depth 3 -E node_modules -E .git -E __pycache__ -E .next -E target -E build -E dist --strip-cwd-prefix 2>/dev/null | head -60
echo '```'
echo ""
echo "## Config Files"
for f in package.json tsconfig.json go.mod Cargo.toml pyproject.toml Dockerfile Makefile next.config.ts angular.json; do
  [ -f "$DIR/$f" ] && echo "- \`$f\`"
done
```
